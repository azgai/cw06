2. TABLE CREATION
--------------------------------------------------
CREATE TABLE employees (
    emp_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    position VARCHAR(100) NOT NULL,
    salary DECIMAL(10,2) NOT NULL
);

--------------------------------------------------
4. SELECT QUERIES
--------------------------------------------------
SELECT * FROM employees;

SELECT emp_id, name, position, salary
FROM employees
ORDER BY salary DESC;

--------------------------------------------------
5. UPDATE EXAMPLE
--------------------------------------------------
UPDATE employees
SET salary = 80000.00
WHERE emp_id = 1;

--------------------------------------------------
6. DELETE EXAMPLE
--------------------------------------------------
DELETE FROM employees
WHERE emp_id = 3;

--------------------------------------------------
7. SAFETY NOTE
--------------------------------------------------
All UPDATE and DELETE queries use WHERE clause
to prevent affecting all rows in the table.
